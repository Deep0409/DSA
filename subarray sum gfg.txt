#include <iostream>

using namespace std;

int main()
{
    int n=5,sum,arr[n],S=15;
    for(int i=0;i<n;i++)
    cin>>arr[i];

    for(int i=0,j=i;i<n,j<n;i++,j++)
    {
        sum=0;
        sum+=arr[j];
        if(sum==S)
        {   
            cout<<i+1<<" "<<j+1<<endl;
            break;
        }
           
    }
    return 0;
}
